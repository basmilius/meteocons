### SVG as image

The simplest approach — import the SVG and render it as an `<img>` tag:

```tsx
import clearDay from '@meteocons/svg/fill/clear-day.svg';

function WeatherIcon() {
    return <img src={clearDay} alt="Clear day" width={64} height={64} />;
}
```

For dynamic icons where the slug is a prop, use a helper to resolve the import:

```tsx
import { useEffect, useState } from 'react';

interface WeatherIconProps {
    slug: string;
    style?: 'fill' | 'flat' | 'line' | 'monochrome';
    size?: number;
}

function WeatherIcon({ slug, style = 'fill', size = 64 }: WeatherIconProps) {
    const [src, setSrc] = useState('');

    useEffect(() => {
        import(`@meteocons/svg/${style}/${slug}.svg`).then(
            (mod) => setSrc(mod.default)
        );
    }, [slug, style]);

    if (!src) {
        return <div style={{ width: size, height: size }} />;
    }

    return <img src={src} alt={slug} width={size} height={size} />;
}

// Usage
<WeatherIcon slug="rain" />
<WeatherIcon slug="snow" style="line" size={48} />
```

### SVG inline

For full CSS control, fetch and inject the SVG markup:

```tsx
import { useEffect, useRef, useState } from 'react';

function InlineSvgIcon({ slug, style = 'fill', size = 64 }: WeatherIconProps) {
    const containerRef = useRef<HTMLDivElement>(null);
    const [svgUrl, setSvgUrl] = useState('');

    useEffect(() => {
        import(`@meteocons/svg/${style}/${slug}.svg`).then(
            (mod) => setSvgUrl(mod.default)
        );
    }, [slug, style]);

    useEffect(() => {
        if (!svgUrl || !containerRef.current) {
            return;
        }

        const controller = new AbortController();

        fetch(svgUrl, { signal: controller.signal })
            .then(res => res.text())
            .then(svg => {
                if (containerRef.current) {
                    containerRef.current.innerHTML = svg;
                }
            });

        return () => controller.abort();
    }, [svgUrl]);

    return <div ref={containerRef} style={{ width: size, height: size }} />;
}
```

**Static import** — the animation JSON is bundled at build time:

```tsx
import Lottie from 'lottie-react';
import rainAnimation from '@meteocons/lottie/fill/rain.json';

function RainIcon() {
    return (
        <Lottie
            animationData={rainAnimation}
            loop
            autoplay
            style={{ width: 64, height: 64 }}
        />
    );
}
```

**Dynamic import** — load the animation at runtime, useful when the icon slug is a prop:

```tsx
import { useEffect, useState } from 'react';
import Lottie from 'lottie-react';

function DynamicWeatherIcon({ slug, style = 'fill', size = 64 }: WeatherIconProps) {
    const [animationData, setAnimationData] = useState(null);

    useEffect(() => {
        import(`@meteocons/lottie/${style}/${slug}.json`).then(
            (mod) => setAnimationData(mod.default)
        );
    }, [slug, style]);

    if (!animationData) {
        return <div style={{ width: size, height: size }} />;
    }

    return (
        <Lottie
            animationData={animationData}
            loop
            autoplay
            style={{ width: size, height: size }}
        />
    );
}
```

### SVG as image

```vue
<script setup lang="ts">
import { computed, defineAsyncComponent, ref, watchEffect } from 'vue';

const props = defineProps<{
    slug: string
    style?: 'fill' | 'flat' | 'line' | 'monochrome'
    size?: number
}>();

const src = ref('');

watchEffect(async () => {
    const mod = await import(`@meteocons/svg/${props.style ?? 'fill'}/${props.slug}.svg`);
    src.value = mod.default;
});
</script>

<template>
    <img
        v-if="src"
        :src="src"
        :alt="slug"
        :width="size ?? 64"
        :height="size ?? 64"
    />
</template>
```

### SVG inline

```vue
<script setup lang="ts">
import { ref, watchEffect } from 'vue';

const props = defineProps<{
    slug: string
    style?: 'fill' | 'flat' | 'line' | 'monochrome'
    size?: number
}>();

const svgMarkup = ref('');

watchEffect(async (onCleanup) => {
    const controller = new AbortController();
    onCleanup(() => controller.abort());

    const mod = await import(`@meteocons/svg/${props.style ?? 'fill'}/${props.slug}.svg`);
    const response = await fetch(mod.default, { signal: controller.signal });
    svgMarkup.value = await response.text();
});
</script>

<template>
    <div
        v-html="svgMarkup"
        :style="{ width: `${size ?? 64}px`, height: `${size ?? 64}px` }"
    />
</template>
```

### Lottie with client islands

Astro is static by default, so Lottie animations need a client-side framework island. Use the Vue or React Lottie component from above with `client:visible` to load the animation only when it scrolls into view:

```astro
---
import WeatherLottieIcon from '../components/WeatherLottieIcon.vue';
---
<WeatherLottieIcon slug="rain" client:visible />
```
